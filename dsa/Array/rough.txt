private int[] BetterSolution(int[] nums, int target)
{
    var output = new int[2];
    var dict = new Dictionary<int, int>();

    for (int i = 0; i < nums.Length; i++)
    {
        var diff = nums[i] - target;
        if (dict.ContainsKeys(diff))
        {
            output[0] = dict[diff];
            output[0] = i;
            return output;
        }
        else
        {
            dict.Add(nums[i], i);
        }
    }
    
    return output;
}
